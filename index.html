async function generateSonnet() { const userTopic = document.getElementById("topicInput").value; const loadingMsg = document.getElementById("loadingMsg"); const outputContainer = document.getElementById("outputContainer"); const poemBox = document.getElementById("poemOutput"); const summaryBox = document.getElementById("summaryOutput"); const submitBtn = document.querySelector("button"); // বাটনটি ধরলাম if(!userTopic) { loadingMsg.innerText = "দয়া করে আগে একটি বিষয় লিখুন!"; loadingMsg.className = "error-msg"; outputContainer.style.display = "none"; return; } // বাটন নিষ্ক্রিয় করা হলো যাতে কেউ বারবার ক্লিক করতে না পারে submitBtn.disabled = true; submitBtn.style.background = "#cccccc"; loadingMsg.innerText = "আপনার সনেট এবং মূলভাব লেখা হচ্ছে, অনুগ্রহ করে অপেক্ষা করুন..."; loadingMsg.className = ""; outputContainer.style.display = "none"; const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=${GEMINI_API_KEY}`; const requestBody = { contents: [{ parts: [{ text: `আপনি একজন প্রখ্যাত কবি। '${userTopic}' বিষয়ের উপর বাংলা ভাষায় একটি চমৎকার সনেট বা চতুর্দশপদী কবিতা লিখুন। প্রথম ৮ লাইন এবং পরের ৬ লাইন আলাদা করে লিখবেন। কবিতা লেখা শেষ হলে ঠিক নিচে '|||' এই চিহ্নটি দেবেন এবং তারপর কবিতার মূলভাব (৩-৪ লাইনে) লিখবেন। ফরম্যাট ঠিক এরকম হবে: [কবিতার ১৪ লাইন] ||| [মূলভাব]` }] }], safetySettings: [ { category: "HARM_CATEGORY_HARASSMENT", threshold: "BLOCK_NONE" }, { category: "HARM_CATEGORY_HATE_SPEECH", threshold: "BLOCK_NONE" }, { category: "HARM_CATEGORY_SEXUALLY_EXPLICIT", threshold: "BLOCK_NONE" }, { category: "HARM_CATEGORY_DANGEROUS_CONTENT", threshold: "BLOCK_NONE" } ] }; try { const response = await fetch(url, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify(requestBody) }); const data = await response.json(); if (!response.ok) { loadingMsg.innerText = `সার্ভার ব্যস্ত বা লিমিট শেষ! দয়া করে ১ মিনিট অপেক্ষা করে আবার চেষ্টা করুন।`; loadingMsg.className = "error-msg"; } else if (data.candidates && data.candidates[0].content) { const fullText = data.candidates[0].content.parts[0].text; const splitText = fullText.split('|||'); const poemText = splitText[0].trim(); const summaryText = splitText.length > 1 ? splitText[1].trim() : "দুঃখিত, মূলভাব তৈরি করা যায়নি।"; poemBox.innerText = poemText; summaryBox.innerText = summaryText; loadingMsg.innerText = ""; outputContainer.style.display = "flex"; } } catch (error) { loadingMsg.innerText = "নেটওয়ার্ক সমস্যা! আপনার ইন্টারনেট বা ভিপিএন (VPN) কানেকশন চেক করুন।"; loadingMsg.className = "error-msg"; } finally { // কাজ শেষ হলে বাটন আবার সচল করা হলো submitBtn.disabled = false; submitBtn.style.background = "#007bff"; } }
